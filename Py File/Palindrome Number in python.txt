Function to check if a number is a palindrome number:
def palin(n):
    """This function checks if a given number is a palindrome number or not."""
    n1 = str(n)
    l1 = list(n1)
    l2 = l1[::-1]
    if l1 == l2:
        return "a Palindrome Number"
    else:
        return "Not a Palindrome Number"
